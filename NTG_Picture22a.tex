\begin{tikzpicture}[>=latex']

% Plot curve C
\draw[black, very thick, use Hobby shortcut] ([out angle=50]a1) .. (a2) .. (a3) .. (a3b) .. (a4) .. (a5) .. (a3) .. (a6) .. ([in angle=240]a7);
